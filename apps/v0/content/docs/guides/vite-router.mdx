# React Router Protection

This guide shows how to protect routes in a Vite + React Router application with better-auth.

## Route Loader Protection

Use React Router loaders for route protection:

```tsx title="src/main.tsx"
import {
  createBrowserRouter,
  RouterProvider,
  redirect,
} from 'react-router-dom'
import { authClient } from './lib/auth-client'

// Auth guard loader
async function requireAuth({ request }: { request: Request }) {
  const session = await authClient.session.get()

  if (!session) {
    const url = new URL(request.url)
    throw redirect('/auth/sign-in?callbackUrl=' + url.pathname)
  }

  return { session }
}

// Guest guard (redirect if logged in)
async function requireGuest() {
  const session = await authClient.session.get()

  if (session) {
    throw redirect('/dashboard')
  }

  return null
}

const router = createBrowserRouter([
  {
    path: '/auth/sign-in',
    element: <SignInPage />,
    loader: requireGuest,
  },
  {
    path: '/dashboard',
    element: <DashboardPage />,
    loader: requireAuth,
  },
])
```

## Protected Route Component

Create a reusable protected route component:

```tsx title="src/components/protected-route.tsx"
import { Navigate, Outlet, useLocation } from 'react-router-dom'
import { useAuth } from '@/hooks/use-auth'

export function ProtectedRoute() {
  const { session, isLoading } = useAuth()
  const location = useLocation()

  if (isLoading) {
    return <div>Loading...</div>
  }

  if (!session) {
    return (
      <Navigate
        to={`/auth/sign-in?callbackUrl=${location.pathname}`}
        replace
      />
    )
  }

  return <Outlet />
}
```

## Auth Hook

Create a custom hook for auth state:

```tsx title="src/hooks/use-auth.tsx"
import { createContext, useContext, useEffect, useState, ReactNode } from 'react'
import { authClient } from '@/lib/auth-client'

type Session = {
  user: {
    id: string
    email: string
    name?: string
  }
}

type AuthContextType = {
  session: Session | null
  isLoading: boolean
  signOut: () => Promise<void>
}

const AuthContext = createContext<AuthContextType | null>(null)

export function AuthProvider({ children }: { children: ReactNode }) {
  const [session, setSession] = useState<Session | null>(null)
  const [isLoading, setIsLoading] = useState(true)

  useEffect(() => {
    async function loadSession() {
      try {
        const data = await authClient.session.get()
        setSession(data)
      } catch {
        setSession(null)
      } finally {
        setIsLoading(false)
      }
    }

    loadSession()
  }, [])

  const signOut = async () => {
    await authClient.signOut()
    setSession(null)
  }

  return (
    <AuthContext.Provider value={{ session, isLoading, signOut }}>
      {children}
    </AuthContext.Provider>
  )
}

export function useAuth() {
  const context = useContext(AuthContext)
  if (!context) {
    throw new Error('useAuth must be used within AuthProvider')
  }
  return context
}
```

## Full Router Setup with Layout Routes

```tsx title="src/main.tsx"
import { StrictMode } from 'react'
import { createRoot } from 'react-dom/client'
import {
  createBrowserRouter,
  RouterProvider,
  redirect,
} from 'react-router-dom'
import { AuthProvider } from '@/hooks/use-auth'
import { authClient } from '@/lib/auth-client'

// Layout with AuthProvider
import RootLayout from './layouts/root'
import DashboardLayout from './layouts/dashboard'

// Pages
import HomePage from './routes/home'
import SignInPage from './routes/auth/sign-in'
import DashboardPage from './routes/dashboard'
import SettingsPage from './routes/dashboard/settings'

const router = createBrowserRouter([
  {
    path: '/',
    element: <RootLayout />,
    children: [
      {
        index: true,
        element: <HomePage />,
      },
      {
        path: 'auth',
        children: [
          {
            path: 'sign-in',
            element: <SignInPage />,
            loader: async () => {
              const session = await authClient.session.get()
              if (session) throw redirect('/dashboard')
              return null
            },
          },
        ],
      },
      {
        path: 'dashboard',
        element: <DashboardLayout />,
        loader: async ({ request }) => {
          const session = await authClient.session.get()
          if (!session) {
            const url = new URL(request.url)
            throw redirect('/auth/sign-in?callbackUrl=' + url.pathname)
          }
          return { session }
        },
        children: [
          {
            index: true,
            element: <DashboardPage />,
          },
          {
            path: 'settings',
            element: <SettingsPage />,
          },
        ],
      },
    ],
  },
])

createRoot(document.getElementById('root')!).render(
  <StrictMode>
    <AuthProvider>
      <RouterProvider router={router} />
    </AuthProvider>
  </StrictMode>
)
```

## Dashboard Layout with Session Data

```tsx title="src/layouts/dashboard.tsx"
import { Outlet, useLoaderData, NavLink } from 'react-router-dom'
import { useAuth } from '@/hooks/use-auth'

export default function DashboardLayout() {
  const { session } = useLoaderData() as { session: any }
  const { signOut } = useAuth()

  return (
    <div className="flex min-h-screen">
      {/* Sidebar */}
      <aside className="w-64 border-r bg-gray-50 p-4">
        <nav className="space-y-2">
          <NavLink
            to="/dashboard"
            end
            className={({ isActive }) =>
              `block rounded px-3 py-2 ${
                isActive ? 'bg-gray-200' : 'hover:bg-gray-100'
              }`
            }
          >
            Dashboard
          </NavLink>
          <NavLink
            to="/dashboard/settings"
            className={({ isActive }) =>
              `block rounded px-3 py-2 ${
                isActive ? 'bg-gray-200' : 'hover:bg-gray-100'
              }`
            }
          >
            Settings
          </NavLink>
        </nav>

        <div className="mt-8 border-t pt-4">
          <p className="text-sm text-gray-600">{session.user.email}</p>
          <button
            onClick={signOut}
            className="mt-2 text-sm text-red-600 hover:underline"
          >
            Sign out
          </button>
        </div>
      </aside>

      {/* Main content */}
      <main className="flex-1 p-8">
        <Outlet />
      </main>
    </div>
  )
}
```

## Role-Based Route Protection

```tsx
async function requireRole(role: string) {
  return async ({ request }: { request: Request }) => {
    const session = await authClient.session.get()

    if (!session) {
      const url = new URL(request.url)
      throw redirect('/auth/sign-in?callbackUrl=' + url.pathname)
    }

    if (session.user.role !== role && session.user.role !== 'admin') {
      throw redirect('/unauthorized')
    }

    return { session }
  }
}

// Usage in router
{
  path: '/admin',
  element: <AdminPage />,
  loader: requireRole('admin'),
}
```

## Best Practices

1. **Use loaders for data fetching and auth checks**
2. **Implement the AuthProvider for global state**
3. **Use redirect() for navigation in loaders**
4. **Keep route protection logic centralized**
5. **Handle loading states gracefully**
